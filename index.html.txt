  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
    import { getAuth, signInWithEmailAndPassword, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-auth.js";
    import { getDatabase, ref, push, remove, onValue, off, set } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js";

    const firebaseConfig = {
      apiKey: "AIzaSyAfQcJ7QJJlZmr_RZRjNyHrAeS217vjawQ",
      authDomain: "geofencing-6271a.firebaseapp.com",
      databaseURL: "https://geofencing-6271a-default-rtdb.firebaseio.com",
      projectId: "geofencing-6271a",
      storageBucket: "geofencing-6271a.appspot.com",
      messagingSenderId: "177087641267",
      appId: "1:177087641267:web:3545bb77073cc5078a22c6"
    };

    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getDatabase(app);

    let currentUser;
    let map;
    let marker;
    let geofenceCircle;
    let activeDeviceRef = null;
    let geofenceRef = null;
    let selectedDeviceId = null;
    let alertTriggered = false;
    let currentRadius = 200;
    let alertSound = document.getElementById("alertSound");

    if (Notification.permission !== "granted") {
      Notification.requestPermission();
    }

    window.login = function () {
      signInWithEmailAndPassword(auth, email.value, password.value)
        .catch(e => alert(e.message));
    };

    window.logout = function () {
      signOut(auth);
    };

    onAuthStateChanged(auth, user => {
      if (user) {
        currentUser = user;
        loginSection.style.display = "none";
        dashboardSection.style.display = "block";
        loadChildren();
      } else {
        loginSection.style.display = "block";
        dashboardSection.style.display = "none";
      }
    });

    window.addChild = function () {
      if (!childName.value || !deviceId.value) {
        alert("Fill all fields");
        return;
      }

      push(ref(db, "parents/" + currentUser.uid + "/children"), {
        name: childName.value,
        deviceId: deviceId.value
      });

      childName.value = "";
      deviceId.value = "";
    };

    function loadChildren() {
      onValue(ref(db, "parents/" + currentUser.uid + "/children"), snap => {
        childrenList.innerHTML = "";
        if (snap.exists()) {
          snap.forEach(child => {
            let data = child.val();
            childrenList.innerHTML += `
          <div class="child-card" onclick="trackChild('${data.deviceId}')">
            👶 ${data.name}
            <button onclick="deleteChild('${child.key}');event.stopPropagation()">❌</button>
          </div>
        `;
          });
        }
      });
    }

    window.deleteChild = function (key) {
      remove(ref(db, "parents/" + currentUser.uid + "/children/" + key));
    };

    window.trackChild = function (deviceId) {

      selectedDeviceId = deviceId;
      alertTriggered = false;

      if (!map) {
        map = L.map('map').setView([20, 78], 5);

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
          attribution: '© OpenStreetMap contributors'
        }).addTo(map);

        map.on("click", function (e) {
          if (!selectedDeviceId) {
            alert("Select a child first");
            return;
          }

          set(ref(db, "geofence/" + selectedDeviceId), {
            centerLat: e.latlng.lat,
            centerLon: e.latlng.lng,
            radius: currentRadius
          });

          alert("Safe zone set successfully!");
        });
      }

      if (activeDeviceRef) off(activeDeviceRef);
      if (geofenceRef) off(geofenceRef);

      activeDeviceRef = ref(db, "devices/" + deviceId);

      onValue(activeDeviceRef, snap => {
        if (snap.exists()) {
          let data = snap.val();
          let lat = parseFloat(data.latitude);
          let lng = parseFloat(data.longitude);

          if (!isNaN(lat) && !isNaN(lng)) {
            if (!marker) {
              marker = L.marker([lat, lng]).addTo(map);
            } else {
              marker.setLatLng([lat, lng]);
            }
            map.setView([lat, lng], 16);
            checkGeofence(lat, lng);
          }
        }
      });

      geofenceRef = ref(db, "geofence/" + deviceId);

      onValue(geofenceRef, snap => {
        if (snap.exists()) {
          let data = snap.val();
          let centerLat = parseFloat(data.centerLat);
          let centerLng = parseFloat(data.centerLon);
          currentRadius = parseFloat(data.radius || 200);

          document.getElementById("radiusSlider").value = currentRadius;
          document.getElementById("radiusValue").innerText = currentRadius;

          if (geofenceCircle) map.removeLayer(geofenceCircle);

          geofenceCircle = L.circle([centerLat, centerLng], {
            color: "green",
            fillColor: "green",
            fillOpacity: 0.3,
            radius: currentRadius
          }).addTo(map);
        }
      });
    };

    document.getElementById("radiusSlider").addEventListener("input", function () {
      currentRadius = parseInt(this.value);
      document.getElementById("radiusValue").innerText = currentRadius;

      if (geofenceCircle && selectedDeviceId) {
        geofenceCircle.setRadius(currentRadius);

        set(ref(db, "geofence/" + selectedDeviceId), {
          centerLat: geofenceCircle.getLatLng().lat,
          centerLon: geofenceCircle.getLatLng().lng,
          radius: currentRadius
        });
      }
    });

    function getDistance(lat1, lon1, lat2, lon2) {
      const R = 6371000;
      const dLat = (lat2 - lat1) * Math.PI / 180;
      const dLon = (lon2 - lon1) * Math.PI / 180;
      const a = Math.sin(dLat / 2) ** 2 +
        Math.cos(lat1 * Math.PI / 180) *
        Math.cos(lat2 * Math.PI / 180) *
        Math.sin(dLon / 2) ** 2;
      return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    }

    function checkGeofence(lat, lng) {

      if (!geofenceCircle) return;

      let center = geofenceCircle.getLatLng();
      let radius = geofenceCircle.getRadius();

      let distance = Math.round(getDistance(lat, lng, center.lat, center.lng));
      document.getElementById("distanceDisplay").innerText = distance;

      if (distance > radius) {

        if (!alertTriggered) {
          alertTriggered = true;

          geofenceCircle.setStyle({ color: "red", fillColor: "red" });
          document.body.classList.add("flash");

          alertSound.play().catch(() => { });

          if (Notification.permission === "granted") {
            new Notification("🚨 Geofence Alert!", {
              body: "Child has exited the safe zone!",
              icon: "https://cdn-icons-png.flaticon.com/512/565/565547.png"
            });
          }

          setTimeout(() => {
            alertSound.pause();
            alertSound.currentTime = 0;
            document.body.classList.remove("flash");
          }, 30000);
        }

      } else {

        alertTriggered = false;

        geofenceCircle.setStyle({ color: "green", fillColor: "green" });
        alertSound.pause();
        alertSound.currentTime = 0;
        document.body.classList.remove("flash");
      }
    }
